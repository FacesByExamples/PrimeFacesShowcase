Editor

Editor is an input component with advanced rich text editing features. Toolbar can be customized with predefined controls using toolbar option and advanced customization can be implemented using toolbarTemplate option.

<h:form id="editorForm">

    <h:panelGrid columns="2">
        <h:outputLabel for="toolbar" value="Toolbar: " />
        <h:selectOneMenu id="toolbar" value="\#{editorBean.toolbar}"
                onchange="document.editorForm.submit()">
            <f:selectItem itemLabel="Complete" itemValue="complete" />
            <f:selectItem itemLabel="Tiny" itemValue="tiny" />
            <f:selectItem itemLabel="Compact" itemValue="compact" />
            <f:selectItem itemLabel="Normal" itemValue="normal" />
            <f:selectItem itemLabel="Maxi" itemValue="maxi" />
        </h:selectOneMenu>
    </h:panelGrid>

    <p:editor value="\#{editorBean.value}" widgetVar="editor"
            toolbar="\#{editorBean.toolbar}" />

    <p:commandButton value="Submit" oncomplete="dlg.show()" onclick="editor.saveHTML()" update="display" />

    <p:dialog header="Content" widgetVar="dlg" showEffect="fade" hideEffect="fade" modal="true">
        <h:outputText id="display" value="\#{editorBean.value}" escape="false" />
    </p:dialog>

</h:form>
                    
package org.primefaces.examples.view;

public class EditorBean {

    private String toolbar = "complete";

	private String value;

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public String getToolbar() {
        return toolbar;
    }

    public void setToolbar(String toolbar) {
        this.toolbar = toolbar;
    }
}
