Tooltip - Configuration

Tooltip has various customization options such as positioning, delay, effects and events.

<h:panelGrid columns="3" cellpadding="5">

    <h:outputLink id="longerDelay" value="#">
        <h:outputText value="Longer" />

    <p:tooltip for="longerDelay" value="Show and hide delays are extended for this tip" showDelay="2000" hideDelay="2000"/>

    <h:outputText value="Focus: " />
    <p:inputText id="focus" />

        <h:outputText value="Fade" />

    <p:tooltip for="fade" value="Fade effect is used by default"
               showEffect="fade" hideEffect="fade" showEffectLength="1000" hideEffectLength="1000"/>

    <h:outputText value="Slide: " />
    <h:outputLink id="slide" value="#">
        <h:outputText value="Slide" />
    </h:outputLink>
    <p:tooltip for="slide" value="This tooltip uses slide effect for the animation"
                        showEffect="slide" hideEffect="slide"
                        showEffectLength="500" hideEffectLength="500"/>
