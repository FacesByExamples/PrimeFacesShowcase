Tooltip - Global

Global tooltip binds to clickable elements with title attributes.

Source

<p:tooltip />
            
<h:panelGrid columns="2" cellpadding="5">

    <h:outputText value="Input: " />
    <p:inputText id="focus" title="Tooltip for an input"/>

    <h:outputText value="Link: " />
    <h:outputLink id="fade" value="#" title="Tooltip for a link">
        <h:outputText value="Fade Effect" />
    </h:outputLink>

    <h:outputText value="Button: " />
    <p:commandButton value="Submit" title="Tooltip for a button" />

</h:panelGrid>
