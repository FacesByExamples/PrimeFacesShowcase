<p:tabView widgetVar="tabs">

	<p:tab title="Default Tab">
		Default Tab Content
	</p:tab>
	
</p:tabView>

<p:dialog header="New Tab" widgetVar="dialog" fixedCenter="true" width="300px"
	effect="fade" effectDuration="0.5">
	<h:form prependId="false">
		<h:panelGrid columns="2">
			<h:outputLabel for="tabTitle" value="Title: " />
			<h:inputText id="tabTitle" />
			
			<h:outputLabel for="tabContent" value="Content: " />
			<h:inputTextarea id="tabContent" style="width:200px"/>	
		</h:panelGrid>
		
		<p:commandButton value="Add" onclick="addTab()"/>
	</h:form>
</p:dialog>
		
<script type="text/javascript">
	function addTab() {
		var tabLabel = document.getElementById('tabTitle').value,
		tabContent = document.getElementById('tabContent').value;

		tabs.addTab(new YAHOO.widget.Tab({label: tabLabel, content: tabContent}));

		//Activate if this is the first tab
		if(tabs.get('tabs').length == 1) {
			tabs.set('activeIndex', 0);		
		}

		dialog.hide();

		return false;
	}

	function removeTab() {
		tabs.removeTab(tabs.get('activeTab'));
	}
</script>
