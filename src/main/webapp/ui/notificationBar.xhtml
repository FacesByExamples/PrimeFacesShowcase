Notification Bar

NotificationBar displays a multipurpose fixed positioned panel for notification either on top or bottom of the page. Any group of JSF components can be placed inside notification bar.

<h:form id="form">
				
	<h:panelGrid columns="3">
		<h:outputLabel for="name" value="Name: *" />
		<p:inputText id="name" value="\#{buttonBean.text}" label="name" required="true" requiredMessage="You must type something!"/>
		<p:commandButton value="Submit" update="bottomPanel" oncomplete="bottomBar.show();"/>
	</h:panelGrid>
	
	<p:notificationBar position="top" effect="slide" widgetVar="topBar" styleClass="top">
		<h:outputText value="What PrimeFaces has so far is just the beginning!" style="color:#FFCC00;font-size:36px;" />
	</p:notificationBar>
	
	<p:notificationBar position="bottom" effect="fade" widgetVar="bottomBar" styleClass="bottom">
		<f:facet name="close">
			<p:graphicImage value="/images/closebar.png" />
		</f:facet>
		
		<h:panelGroup id="bottomPanel" style="margin-right:200px;">
			<p:messages />
			<h:outputText value="Welcome \#{buttonBean.text}" rendered="\#{not empty buttonBean.text}"/>
		</h:panelGroup>
	</p:notificationBar>
</h:form>
